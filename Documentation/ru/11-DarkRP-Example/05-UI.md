## DarkRPHud.razor — Игровой HUD

```razor
@* code/UI/Hud/DarkRPHud.razor *@

@using Sandbox;
@using Sandbox.UI;
@inherits PanelComponent
@namespace DarkRP.UI

<root>
    @* === ЗДОРОВЬЕ И БРОНЯ (левый низ) === *@
    <div class="bottom-left">
        <div class="health-bar">
            <div class="bar-fill health" style="width: @(HealthPercent)%" />
            <div class="bar-text">❤️ @((int)Health)</div>
        </div>
        @if ( Armor > 0 )
        {
            <div class="health-bar">
                <div class="bar-fill armor" style="width: @(Armor)%" />
                <div class="bar-text">🛡️ @Armor</div>
            </div>
        }
    </div>
    
    @* === ДЕНЬГИ (левый верх) === *@
    <div class="top-left">
        <div class="money">$@Money</div>
        <div class="job" style="color: @JobColorHex">@JobName</div>
    </div>
    
    @* === ПЕРЕКРЕСТЬЕ === *@
    <div class="crosshair">+</div>
    
    @* === ПОДСКАЗКА ВЗАИМОДЕЙСТВИЯ (центр) === *@
    @if ( !string.IsNullOrEmpty(InteractText) )
    {
        <div class="interact-hint">
            <span class="key">E</span>
            <span class="text">@InteractText</span>
        </div>
    }
    
    @* === ИНФОРМАЦИЯ О ЦЕЛИ (центр, при наведении на игрока) === *@
    @if ( !string.IsNullOrEmpty(TargetName) )
    {
        <div class="target-info">
            <div class="target-name" style="color: @TargetJobColorHex">@TargetName</div>
            <div class="target-job">@TargetJobName</div>
            @if ( TargetIsWanted )
            {
                <div class="wanted-badge">⚠️ РАЗЫСКИВАЕТСЯ</div>
            }
        </div>
    }
</root>

@code
{
    // Данные из компонента игрока
    [Property] public float Health { get; set; } = 100f;
    [Property] public float MaxHealth { get; set; } = 100f;
    [Property] public int Armor { get; set; } = 0;
    [Property] public int Money { get; set; } = 500;
    [Property] public string JobName { get; set; } = "Гражданин";
    [Property] public Color JobColor { get; set; } = Color.White;
    
    // Подсказка взаимодействия
    public string InteractText { get; set; }
    
    // Информация о цели
    public string TargetName { get; set; }
    public string TargetJobName { get; set; }
    public Color TargetJobColor { get; set; }
    public bool TargetIsWanted { get; set; }
    
    // Вычисляемые свойства
    float HealthPercent => (Health / MaxHealth * 100f).Clamp( 0, 100 );
    string JobColorHex => $"rgba({(int)(JobColor.r*255)}, {(int)(JobColor.g*255)}, {(int)(JobColor.b*255)}, 1)";
    string TargetJobColorHex => $"rgba({(int)(TargetJobColor.r*255)}, {(int)(TargetJobColor.g*255)}, {(int)(TargetJobColor.b*255)}, 1)";
    
    protected override void OnUpdate()
    {
        // Обновляем данные из компонента игрока
        var player = Scene.GetAllComponents<DarkRPPlayer>()
            .FirstOrDefault( p => !p.IsProxy );
        
        if ( player != null )
        {
            Health = player.Health;
            MaxHealth = player.MaxHealth;
            Armor = player.Armor;
            Money = player.Money;
            JobName = player.JobName;
            JobColor = player.JobColor;
        }
        
        // Проверяем, на что смотрит игрок
        UpdateLookTarget( player );
    }
    
    void UpdateLookTarget( DarkRPPlayer player )
    {
        InteractText = null;
        TargetName = null;
        TargetJobName = null;
        TargetIsWanted = false;
        
        if ( player == null ) return;
        
        var eyePos = player.CameraObject?.WorldPosition ?? player.WorldPosition;
        var eyeDir = player.EyeAngles.ToRotation().Forward;
        
        var tr = Scene.Trace
            .Ray( eyePos, eyePos + eyeDir * 300 )
            .IgnoreGameObject( player.GameObject )
            .Run();
        
        if ( !tr.Hit || tr.GameObject == null ) return;
        
        // Проверяем интерактивные объекты
        var interactable = tr.GameObject.Components.Get<IInteractable>();
        if ( interactable != null )
        {
            InteractText = interactable.InteractText;
        }
        
        // Проверяем игроков
        var targetPlayer = tr.GameObject.Components.GetInParent<DarkRPPlayer>();
        if ( targetPlayer != null )
        {
            TargetName = targetPlayer.PlayerName;
            TargetJobName = targetPlayer.JobName;
            TargetJobColor = targetPlayer.JobColor;
            TargetIsWanted = targetPlayer.IsWanted;
        }
    }
    
    protected override int BuildHash() => System.HashCode.Combine(
        Health, Armor, Money, JobName, InteractText, TargetName
    );
}
```

## JobMenu.razor — Меню выбора профессии

```razor
@* code/UI/JobMenu/JobMenu.razor *@

@using Sandbox;
@using Sandbox.UI;
@using System.Linq;
@inherits PanelComponent
@namespace DarkRP.UI

<root class="@(IsVisible ? "visible" : "hidden")">
    <div class="menu-overlay">
        <div class="menu-window">
            <div class="menu-header">
                <h1>Выбор профессии</h1>
                <button class="close-btn" onclick="@Close">✕</button>
            </div>
            
            <div class="menu-body">
                @* Категории профессий *@
                <div class="categories">
                    @foreach ( var category in Categories )
                    {
                        <button class="category-btn @(category == SelectedCategory ? "active" : "")"
                                onclick="@(() => SelectedCategory = category)">
                            @category
                        </button>
                    }
                </div>
                
                @* Список профессий *@
                <div class="job-list">
                    @foreach ( var job in FilteredJobs )
                    {
                        <div class="job-card @(job.JobName == CurrentJobName ? "current" : "")"
                             onclick="@(() => SelectJob(job))">
                            <div class="job-color" style="background-color: rgba(@((int)(job.JobColor.r*255)), @((int)(job.JobColor.g*255)), @((int)(job.JobColor.b*255)), 1)" />
                            <div class="job-info">
                                <div class="job-name">@job.JobName</div>
                                <div class="job-desc">@job.Description</div>
                                <div class="job-stats">
                                    <span>💰 $@job.Salary/мин</span>
                                    @if ( job.MaxPlayers > 0 )
                                    {
                                        <span>👥 @GetJobCount(job.JobName)/@job.MaxPlayers</span>
                                    }
                                    @if ( job.CanUseWeapons )
                                    {
                                        <span>🔫</span>
                                    }
                                    @if ( job.CanArrest )
                                    {
                                        <span>🚔</span>
                                    }
                                </div>
                            </div>
                        </div>
                    }
                </div>
            </div>
        </div>
    </div>
</root>

@code
{
    public bool IsVisible { get; set; } = false;
    public string SelectedCategory { get; set; } = "Гражданские";
    public string CurrentJobName { get; set; } = "Гражданин";
    
    // Данные (заполняются из JobSystem)
    public List<JobDefinition> AllJobs { get; set; } = new();
    
    public List<string> Categories => AllJobs
        .Select( j => j.Category )
        .Distinct()
        .ToList();
    
    public IEnumerable<JobDefinition> FilteredJobs => AllJobs
        .Where( j => j.Category == SelectedCategory );
    
    protected override void OnUpdate()
    {
        // Открываем по F4
        if ( Input.Pressed( "Score" ) ) // или свой кастомный input
        {
            IsVisible = !IsVisible;
        }
    }
    
    void SelectJob( JobDefinition job )
    {
        // Находим JobSystem и назначаем профессию
        var jobSystem = Scene.GetAllComponents<JobSystem>().FirstOrDefault();
        var player = Scene.GetAllComponents<DarkRPPlayer>()
            .FirstOrDefault( p => !p.IsProxy );
        
        if ( jobSystem != null && player != null )
        {
            jobSystem.AssignJob( player, job.JobName );
            CurrentJobName = job.JobName;
            IsVisible = false;
        }
    }
    
    int GetJobCount( string jobName )
    {
        return Scene.GetAllComponents<DarkRPPlayer>()
            .Count( p => p.JobName == jobName );
    }
    
    void Close()
    {
        IsVisible = false;
    }
    
    protected override int BuildHash() => System.HashCode.Combine(
        IsVisible, SelectedCategory, CurrentJobName, AllJobs.Count
    );
}
```
